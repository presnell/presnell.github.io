data_dir <- "~/Personal/Sports/MBL/2023/Data"
xl_file <- "MBL2_2023.xlsx"
rds_file <- "MBL2_2023.rds"

r_script_dir <- "~/Personal/Sports/MBL/2023/R-scripts"
data_script <- "read_mbl_results.R"

xl_file <- file.path(data_dir, xl_file)
rds_file <- file.path(data_dir, rds_file)
data_script <- file.path(r_script_dir, data_script)

library(tidyverse)

mtimes <- file.info(xl_file, rds_file)$mtime
names(mtimes) <- c("src", "sav")

if (mtimes["src"] > mtimes["sav"]) {
    source(data_script)
    saveRDS(mbl, rds_file)
} else {
    mbl <- readRDS(rds_file)
}

Code 
mbl <- mbl |>
    mutate(
        result = (units_won == wager) - (units_won == 0) + 2,
        result = ifelse(is.na(result), 4, result),
        result = c("lose", "push", "win", NA)[result])
Code 
mbl_by_moron_week <- mbl |>
    summarize(
        wins = sum(result == "win"),
        losses = sum(result == "lose"),
        pushes = sum(result == "push"),
        units_won = sum(units_won),
        .by = c(moron, week, mia)
    ) |>
    ## Handle MIA weeks with tidyr::fill().
    arrange(moron, week) |>    # Making sure that observations are ordered correctly.
    group_by(moron) |>         # Grouping means filling done only within group (by moron).
    fill(units_won, .direction = "up") |>
    ungroup() |>
    filter(!is.na(units_won)) |>
    mutate(season_total = cumsum(units_won), .by = moron)
